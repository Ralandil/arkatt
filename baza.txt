#var total_enemy_count {0};

#alias {bindujwroga} {
    #format lwrog {%l} {%1};
    #foreach {$obda2[]} {vob} {
      #format lname {%l} {$obda2[$vob][desc]};
      #if {"$lwrog" == "$lname"} {
	#line substitute {variables} {#alias {$lname} {nazabij ob_$vob}};
	#if {$total_enemy_count == 0} {#var asd @set_quickbind_F1{$lname}};
	#if {$total_enemy_count == 1} {#var asd @set_quickbind_F2{$lname}};
	#if {$total_enemy_count == 2} {#var asd @set_quickbind_F3{$lname}};
	#if {$total_enemy_count == 3} {#var asd @set_quickbind_F4{$lname}};
	#if {$total_enemy_count == 4} {#var asd @set_quickbind_F5{$lname}};
	#if {$total_enemy_count == 5} {#var asd @set_quickbind_F6{$lname}};
	#if {$total_enemy_count == 6} {#var asd @set_quickbind_F7{$lname}};
	#if {$total_enemy_count == 7} {#var asd @set_quickbind_F8{$lname}};
	#if {$total_enemy_count == 8} {#var asd @set_quickbind_F9{$lname}};
	#if {$total_enemy_count == 9} {#var asd @set_quickbind_F10{$lname}};
	#if {$total_enemy_count > 9} {#sh zbindowano maxymalna ilosc wrogow, uzyj komendy xxx};
	#math {total_enemy_count} {$total_enemy_count + 1};
      };
    };
}
